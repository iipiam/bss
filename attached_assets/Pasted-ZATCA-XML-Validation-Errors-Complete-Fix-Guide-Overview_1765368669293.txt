ERROR 1: SIGNATURE_ERROR - "wrong invoiceCertificate"

What This Means

Your digital signature certificate is incorrect or doesn't match your CCSID.

Root Causes

1.
Using the wrong private key

2.
Private key doesn't match your CCSID certificate

3.
Certificate has expired

4.
Certificate format is incorrect

How to Fix in Replit

Step 1: Verify Your Certificate

JavaScript


// Verify certificate matches CCSID
const crypto = require('crypto');
const fs = require('fs');

function verifyCertificate(privateKeyPath, ccsid) {
  try {
    const privateKey = fs.readFileSync(privateKeyPath, 'utf8');
    
    // Check if private key is valid PEM format
    if (!privateKey.includes('BEGIN PRIVATE KEY') && 
        !privateKey.includes('BEGIN RSA PRIVATE KEY')) {
      console.error('ERROR: Invalid private key format');
      return false;
    }
    
    console.log('✓ Private key format is valid');
    return true;
  } catch (error) {
    console.error('ERROR: Cannot read private key file:', error.message);
    return false;
  }
}

// Test it
verifyCertificate('./private_key.pem', process.env.ZATCA_CCSID);

// CORRECT: Use the private key that matches your CCSID
function signInvoiceCorrectly(invoiceXML, privateKeyPath) {

ERROR 2: QR_CODE_ERROR - "Invalid QR code format"

What This Means

Your QR code data is not in the correct ZATCA format.

ZATCA QR Code Format Requirements

The QR code must contain exactly these fields in this exact order:

Plain Text


[Seller Name]: # "| [Seller TIN] | [Invoice Timestamp] | [Invoice Total] | [VAT Amount] | [Invoice Hash] | [Digital Signature]"


How to Fix in Replit

Step 1: Create Correct QR Code Data

JavaScript


// CORRECT QR Code Format
function generateCorrectQRCodeData(invoiceData, invoiceHash, signature) {
  // Format: All fields separated by pipe (|)
  // Order is CRITICAL
  
  const qrData = [
    invoiceData.seller.name,           // Field 1: Seller Name
    invoiceData.seller.tin,            // Field 2: Seller TIN (10 digits)
    invoiceData.issueDate,             // Field 3: Timestamp (ISO 8601)
    invoiceData.totalAmount.toFixed(2), // Field 4: Invoice Total (with 2 decimals)
    invoiceData.taxAmount.toFixed(2),  // Field 5: VAT Amount (with 2 decimals)
    invoiceHash,                       // Field 6: Invoice Hash (SHA-256 hex)
    signature                          // Field 7: Digital Signature (base64)
  ].join('|');
  
  return qrData;
}

// Example output:
// "Tech Store Saudi|1234567890|2024-01-15T10:30:00|230.00|30.00|abc123def456...|xyz789..."


Step 2: Encode QR Code Correctly

JavaScript


const QRCode = require('qrcode');

async function generateQRCodeImage(qrData) {
  try {
    // Generate QR code as data URL
    const qrCodeImage = await QRCode.toDataURL(qrData, {

      type: 'image/png',
      quality: 0.92,
      margin: 1,
      width: 200,
      color: {
        dark: '#000000',
        light: '#FFFFFF'
      }
    });
    
    return qrCodeImage;
  } catch (error) {
    console.error('QR Code generation failed:', error);
    throw error;
  }

Step 3: Validate QR Code Data Before Generation

JavaScript


function validateQRCodeData(qrData) {
  const parts = qrData.split('|');
  
  // Must have exactly 7 parts
  if (parts.length !== 7) {
    console.error(`ERROR: QR code must have 7 fields, got ${parts.length}`);
    return false;
  }
  
  const [sellerName, tin, timestamp, total, vat, hash, sig] = parts;
  
  // Validate each field
  if (!sellerName) {
    console.error('ERROR: Seller name is empty');
    return false;
  }
  
  if (tin.length !== 10 || isNaN(tin)) {
    console.error('ERROR: TIN must be exactly 10 digits');
    return false;
  }
  
  if (!timestamp.includes('T')) {
    console.error('ERROR: Timestamp must be ISO 8601 format (YYYY-MM-DDTHH:MM:SS)');
    return false;
  }
  
  if (isNaN(parseFloat(total))) {
    console.error('ERROR: Total amount must be a number');
    return false;
  }
  
  if (isNaN(parseFloat(vat))) {
    console.error('ERROR: VAT amount must be a number');
    return false;
  }
  
  if (hash.length < 32) {
    console.error('ERROR: Hash appears too short (should be SHA-256)');
    return false;
  }
  
  if (!sig || sig.length < 10) {
    console.error('ERROR: Signature is missing or too short');
    return false;
  }
  
  console.log('✓ QR code data format is valid');
  return true;
}


Step 4: Complete QR Code Generation Function

JavaScript


async function generateCompleteQRCode(invoiceData, invoiceHash, signature) {
  // Step 1: Create QR data in correct format
  const qrData = generateCorrectQRCodeData(invoiceData, invoiceHash, signature);
  
  // Step 2: Validate QR data
  if (!validateQRCodeData(qrData)) {
    throw new Error('QR code data validation failed');
  }
  
  // Step 3: Generate QR code image
  const qrCodeImage = await generateQRCodeImage(qrData);
  
  return {
    data: qrData,
    image: qrCodeImage
  };
}

Step 3: Generate Signature with Error Handling

JavaScript


function generateSignatureWithErrorHandling(invoiceXML, privateKeyPath) {
  try {
    // Validate inputs
    validateInvoiceXML(invoiceXML);
    const privateKey = loadPrivateKey(privateKeyPath);
    
    // Generate signature
    const sign = crypto.createSign('sha256');
    sign.update(invoiceXML);
    
    const signature = sign.sign(privateKey, 'base64');
    
    // Verify signature is not null

      throw new Error('Signature generation returned empty result');

    console.error('ERROR: Signature generation failed:', error.message);
    throw error;
  }
}


Step 4: Embed Signature in XML

JavaScript


function embedSignatureInXML(invoiceXML, signature) {
  // Add signature to XML structure
  const signatureElement = `
    <ext:UBLExtensions>
      <ext:UBLExtension>
        <ext:ExtensionContent>
          <sig:UBLDocumentSignatures>
            <sig:SignatureInformation>
              <sig:id>urn:oasis:names:specification:ubl:signature:Invoice</sig:id>
              <sig:ReferencedSignatureID>urn:oasis:names:specification:ubl:signature:Invoice</sig:ReferencedSignatureID>
              <sig:SignatureMethod>urn:oasis:names:specification:ubl:dsig:enveloped:xades</sig:SignatureMethod>
              <sig:SignatureValue>${signature}</sig:SignatureValue>
            </sig:SignatureInformation>
          </sig:UBLDocumentSignatures>
        </ext:ExtensionContent>
      </ext:UBLExtension>
    </ext:UBLExtensions>
  `;
  
  // Insert after XML declaration and before Invoice element
  const insertPoint = invoiceXML.indexOf('<Invoice');
  const newXML = invoiceXML.substring(0, insertPoint) + 
                 signatureElement + 
                 invoiceXML.substring(insertPoint);
  
  return newXML;
}

WARNING 2 & 3: BR-CO-10 & BR-CO-13 - "Sum calculation warnings"

What This Means

Your invoice totals don't match the line items sum.

How to Fix in Replit

Step 1: Verify Calculation Logic

JavaScript


function calculateInvoiceTotals(lineItems) {
  // Calculate subtotal (sum of all line items before tax)
  const subtotal = lineItems.reduce((sum, item) => {
    return sum + (item.quantity * item.unitPrice);
  }, 0);
  
  // Calculate tax (15% of subtotal)
  const taxAmount = subtotal * 0.15;
  
  // Calculate total (subtotal + tax)
  const totalAmount = subtotal + taxAmount;
  
  return {
    subtotal: parseFloat(subtotal.toFixed(2)),
    taxAmount: parseFloat(taxAmount.toFixed(2)),
    totalAmount: parseFloat(totalAmount.toFixed(2))
  };
}

// Example:
const items = [
  { quantity: 2, unitPrice: 100 }
];
const totals = calculateInvoiceTotals(items);
console.log(totals);
// Output: { subtotal: 200, taxAmount: 30, totalAmount: 230 }


Step 2: Ensure XML Totals Match Calculations

JavaScript


function validateInvoiceTotals(invoiceData) {
  // Calculate what totals should be
  const calculated = calculateInvoiceTotals(invoiceData.lineItems);
  
  // Compare with invoice data
  if (Math.abs(calculated.subtotal - invoiceData.subtotal) > 0.01) {
    console.error('ERROR: Subtotal mismatch');
    console.error('Expected:', calculated.subtotal);
    console.error('Got:', invoiceData.subtotal);
    return false;
  }
  
  if (Math.abs(calculated.taxAmount - invoiceData.taxAmount) > 0.01) {
    console.error('ERROR: Tax amount mismatch');
    console.error('Expected:', calculated.taxAmount);
    console.error('Got:', invoiceData.taxAmount);
    return false;
  }
  
  if (Math.abs(calculated.totalAmount - invoiceData.totalAmount) > 0.01) {
    console.error('ERROR: Total amount mismatch');
    console.error('Expected:', calculated.totalAmount);
    console.error('Got:', invoiceData.totalAmount);
    return false;
  }
  
  console.log('✓ All invoice totals are correct');
  return true;
}

function validateVATBreakdown(invoiceData) {

  const standardVATTotal = invoiceData.lineItems
    .filter(item => item.taxRate === 0.15)
    .reduce((sum, item) => sum + (item.quantity * item.unitPrice), 0);

  // Compare with invoice tax amount
  if (Math.abs(expectedTax - invoiceData.taxAmount) > 0.01) {
    console.error('ERROR: VAT breakdown mismatch');

    console.error('Invoice tax:', invoiceData.taxAmount);

  return true;
}

            <sig:SignatureInformation>
              <sig:id>urn:oasis:names:specification:ubl:signature:1</sig:id>
              <sig:ReferencedSignatureID>urn:oasis:names:specification:ubl:signature:Invoice</sig:ReferencedSignatureID>
              <sig:SignatureMethod>urn:oasis:names:specification:ubl:dsig:enveloped:xades</sig:SignatureMethod>
              <sig:SignatureValue>${signature}</sig:SignatureValue>
            </sig:SignatureInformation>
          </sig:UBLDocumentSignatures>
        </ext:ExtensionContent>
      </ext:UBLExtension>
    </ext:UBLExtensions>`;

    const insertPoint = xmlContent.indexOf('<Invoice');
    return xmlContent.substring(0, insertPoint) + signatureElement + xmlContent.substring(insertPoint);
  }

  // Main function: Generate complete invoice
  async generateCompleteInvoice(invoiceData) {
    try {
      // Validate
      this.validateInvoiceData(invoiceData);

      // Generate XML
      let xml = this.generateXML(invoiceData);

      // Generate hash
      const hash = this.generateHash(xml);

      // Generate signature
      const signature = this.generateSignature(xml);

      // Embed signature in XML
      xml = this.embedSignatureInXML(xml, signature);

      // Generate QR code
      const qr = await this.generateQRCode(invoiceData, hash, signature);

      console.log('✓ Invoice generated successfully');

      return {
        xml: xml,
        hash: hash,
        signature: signature,
        qrCode: qr,
        totals: this.calculateTotals(invoiceData.lineItems)
      };
    } catch (error) {
      console.error('ERROR: Invoice generation failed:', error.message);
      throw error;
    }
  }
}

module.exports = InvoiceGenerator;
